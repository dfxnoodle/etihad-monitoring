    # Etihad Rail Monitoring Dashboard

    # Frontend Static Files
    location /monitoring/ {
        # Update this path to actual frontend/dist folder
        alias /mnt/data/etihad-monitoring/frontend/dist/;
        try_files $uri $uri/ /monitoring/index.html;
    }

    # Backend API Proxy
    location /monitoring/api/ {
        # Proxy to the FastAPI backend running on Unix socket
        proxy_pass http://unix:/tmp/etihad-monitoring.sock:/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }
